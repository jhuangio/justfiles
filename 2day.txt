import {DateRange} from '../../models/date-range.model';

/**
 * Subline that has access to
 */
export class SublineAccess {
  readonly id: string;
  readonly name: string;
  readonly effectiveDateRange: DateRange;

  constructor (sublineAccessJson: any) {
    this.id = sublineAccessJson.id;
    this.name = sublineAccessJson.name;
    if (sublineAccessJson.effectiveDateRange) {
      this.effectiveDateRange = new DateRange(new Date(sublineAccessJson.effectiveDateRange.startDate), new Date(sublineAccessJson.effectiveDateRange.endDate));
    }
  }
}
